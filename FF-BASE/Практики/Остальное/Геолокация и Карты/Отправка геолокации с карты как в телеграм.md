class CustomMapPicker extends StatefulWidget {
  const CustomMapPicker({
    Key? key,
    this.width,
    this.height,
  }) : super(key: key);

  final double? width;
  final double? height;

  @override
  _CustomMapPickerState createState() => _CustomMapPickerState();
}

class _CustomMapPickerState extends State<CustomMapPicker> {
  late gmaps.GoogleMapController mapController;
  late gmaps.LatLng centerPosition;
  String address = 'Выберите местоположение';

  @override
  void initState() {
    super.initState();
    centerPosition = gmaps.LatLng(34.052235,
        -118.243683); // Пример: Лос-Анджелес или текущее местоположение пользователя
  }

  void _onMapCreated(gmaps.GoogleMapController controller) {
    mapController = controller;
  }

  void _onCameraMove(gmaps.CameraPosition position) {
    centerPosition = position.target;
  }

  Future<String> getAddressFromLatLng(double latitude, double longitude) async {
    final apiKey =
        'YUOAPIGOOGLEMAP'; // Замените на ваш API ключ
    final url = Uri.parse(
        'https://maps.googleapis.com/maps/api/geocode/json?latlng=$latitude,$longitude&key=$apiKey');

    try {
      final response = await http.get(url);
      if (response.statusCode == 200) {
        final data = json.decode(response.body);
        if (data['results'].isNotEmpty) {
          // Получаем полный адрес из первого результата
          return data['results'][0]['formatted_address'];
        } else {
          return 'No address available at this location.';
        }
      } else {
        return 'Failed to fetch address: ${response.reasonPhrase}';
      }
    } catch (e) {
      return 'Failed to fetch address: $e';
    }
  }

  void _onCameraIdle() async {
    final fetchedAddress = await getAddressFromLatLng(
        centerPosition.latitude, centerPosition.longitude);
    setState(() {
      address = fetchedAddress; // Обновите адрес в состоянии виджета
    });
    // Обновляем адрес в FFAppState()
    Provider.of<FFAppState>(context, listen: false).Address = fetchedAddress;
  }

  @override
  Widget build(BuildContext context) {
    return Container(
      width: widget.width ?? MediaQuery.of(context).size.width,
      height: widget.height ?? MediaQuery.of(context).size.height,
      child: Stack(
        children: [
          gmaps.GoogleMap(
            onMapCreated: _onMapCreated,
            onCameraMove: _onCameraMove,
            onCameraIdle: _onCameraIdle,
            initialCameraPosition: gmaps.CameraPosition(
              target: centerPosition,
              zoom: 11.0,
            ),
          ),
          Center(
            child: Icon(Icons.location_pin, size: 50),
          ),
          Positioned(
            bottom: 0,
            left: 0,
            right: 0,
            child: Container(
              padding: EdgeInsets.symmetric(horizontal: 15, vertical: 10),
              color: Colors.white,
              child: Text(address, textAlign: TextAlign.center),
            ),
          )
        ],
      ),
    );
  }
}
